Checkboxes


Checkboxes are used to check the options, they help to collect information from the user. See examples below:


Default Checkboxes



<!-- UNCHECKED -->
<label class="checkbox">
<input type="checkbox">
<span>Unchecked</span>
</label>
<!-- /UNCHECKED -->

<!-- CHECKED -->
<label class="checkbox">
<input type="checkbox" checked>
<span>Checked</span>
</label>
<!-- /CHECKED -->

<!-- INDETERMINATE -->
<div class="indeterminate">
<label class="checkbox">
<input type="checkbox">
<span>Indeterminate</span>
</label>
</div>
<!-- INDETERMINATE -->
Create "indeterminate" checkbox by surrounding it <div> with class indeterminate


Disabled



<!-- UNCHECKED DISABLED -->
<label class="checkbox">
<input type="checkbox" disabled>
<span>Unchecked</span>
</label>
<!-- /UNCHECKED DISABLED -->

<!-- CHECKED DISABLED -->
<label class="checkbox">
<input type="checkbox" checked disabled>
<span>Checked</span>
</label>
<!-- /CHECKED DISABLED -->

<!-- INDETERMINATE DISABLED -->
<div class="indeterminate">
<label class="checkbox">
<input type="checkbox" disabled>
<span>Indeterminate</span>
</label>
</div>
<!-- INDETERMINATE DISABLED -->


<!-- WITHOUT FILL UNCHECKED DISABLED -->
<label class="checkbox">
<input type="checkbox" class="not-fill" disabled>
<span>Unchecked</span>
</label>
<!-- /WITHOUT FILL UNCHECKED DISABLED -->

<!-- WITHOUT FILL CHECKED DISABLED -->
<label class="checkbox">
<input type="checkbox" class="not-fill" checked disabled>
<span>Checked</span>
</label>
<!-- /WITHOUT FILL CHECKED DISABLED -->

<!-- WITHOUT FILL INDETERMINATE DISABLED -->
<div class="indeterminate">
<label class="checkbox">
<input type="checkbox" class="not-fill" disabled >
<span>Indeterminate</span>
</label>
</div>
<!-- WITHOUT FILL INDETERMINATE DISABLED -->

Without Fill


<!-- WITHOUT FILL UNCHECKED -->
<label class="checkbox">
<input type="checkbox" class="not-fill">
<span>Unchecked</span>
</label>
<!-- /WITHOUT FILL UNCHECKED -->

<!-- WITHOUT FILL CHECKED -->
<label class="checkbox">
<input type="checkbox" checked class="not-fill">
<span>Checked</span>
</label>
<!-- /WITHOUT FILL CHECKED -->

<!-- WITHOUT FILL INDETERMINATE -->
<div class="indeterminate">
<label class="checkbox">
<input type="checkbox" class="not-fill">
<span>Indeterminate</span>
</label>
</div>
<!-- WITHOUT FILL INDETERMINATE -->
Create checkbox "without fill" by adding to input not-fill classs.
