Switches


Switches are special checkboxes that are used to enable/disable options. See examples below: 


<!-- SWITCH -->
<div class="control">
<input type="checkbox" id="unchecked" class="control-switch" />
<label for="unchecked"></label>
</div>
<!-- /SWITCH -->

<!-- CHECKED SWITCH -->
<div class="control">
<input type="checkbox" id="checked" class="control-switch" checked/>
<label for="checked"></label>
</div>
<!-- /CHECKED SWITCH -->



Disabled



<!-- DISABLED SWITCH -->
<div class="control">
<input type="checkbox" id="unchecked-disabled" class="control-switch" disabled/>
<label for="unchecked-disabled"></label>
</div>
<!-- /DISABLED SWITCH -->

<!-- DISABLED AND CHECKED SWITCH -->
<div class="control">
<input type="checkbox" id="checked-disabled" class="control-switch" checked disabled/>
<label for="checked-disabled"></label>
</div>
<!-- /DISABLED AND CHECKED SWITCH -->
