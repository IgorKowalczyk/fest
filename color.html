Use background color adding COLOR-HUEclass. You can add the text color usingtext-COLOR-HUEclass.

<!-- BLUE-GREY-900 (BACKGROUND)-->
<div class="blue-grey-900">blue-grey-900</div>
<!-- /BLUE-GREY-900 (BACKGROUND)-->

<!-- BLUE-GREY-900 (TEXT)-->
<div class="text-blue-grey-900">text-blue-grey-900</div>
<!-- /BLUE-GREY-900 (TEXT)-->
