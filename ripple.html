Ripple


The ripple effect improves the look of the buttons even more, making them really pretty. Add this effect by adding classes ripple and ripple-effect. See examples below:



For more examples look at buttons 

<!-- SMALL -->
<button class="btn-small ripple ripple-effect">
CSS</button>
<!-- /SMALL -->

<!-- MEDIUM -->
<button class="btn-medium ripple ripple-effect">
CSS</button>
<!-- /MEDIUM -->

<!-- BIG -->
<button class="btn-big ripple ripple-effect">
CSS</button>
<!-- /BIG -->

Order Colors



<!-- AQUA RIPPLE -->
<button class="btn-medium btn-text ripple ripple-effect ripple-aqua">.ripple-aqua</button>
<!-- /AQUA RIPPLE -->

<!-- RED RIPPLE -->
<button class="btn-medium btn-text ripple ripple-effect ripple-red">.ripple-red</button>
<!-- /RED RIPPLE -->

<!-- LIGHT RIPPLE -->
<button class="btn-medium btn-text ripple ripple-effect ripple-light">.ripple-light</button>
<!-- /LIGHT RIPPLE -->

<!-- YELLOW RIPPLE -->
<button class="btn-medium btn-text ripple ripple-effect ripple-yellow">.ripple-yellow</button>
<!-- /YELLOW RIPPLE -->

<!-- ORANGE RIPPLE -->
<button class="btn-medium btn-text ripple ripple-effect ripple-orange">.ripple-orange</button>
<!-- /ORANGE RIPPLE -->

<!-- PURPLE RIPPLE -->
<button class="btn-medium btn-text ripple ripple-effect ripple-purple">.ripple-purple</button>
<!-- /PURPLE RIPPLE -->

<!-- GREEN RIPPLE -->
<div class="align-center">
<button class="btn-medium btn-text ripple ripple-effect ripple-green">.ripple-green</button>
<!-- /GREEN RIPPLE -->

Custom


/* IN .CSS OR IN <STYLE></STYLE> */

/* LIME RIPPLE */

.ripple-effect.ripple-pink .ink {
/* Pink-300 */
background: rgba(241, 114, 157);
/* Box shadow for better results */
box-shadow: 0 0px 10px 0 rgba(241, 114, 157, 0.5);
}

/* /LIME RIPPLE */

/* /IN .CSS OR IN <STYLE></STYLE> */
