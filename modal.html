<!-- DEFAULT MODAL -->

<!-- Modal open button -->

<button class="btn-big ripple ripple-effect modal-open" data-popup-trigger="one">Default Modal</button>

<!-- /Modal open button -->

<!-- Modal body -->

<div class="modal" data-popup-modal="one">
<div class="modal-close"></div>
<h5>Modal Header</h5>
<p>Modal Body</p>
</div>

<!-- /Modal body -->

<!-- /DEFAULT MODAL -->
Note: data-popup-trigger="TEXT" and data-popup-modal="TEXT" will be the same. Create the modal open button by adding modal-open class.
