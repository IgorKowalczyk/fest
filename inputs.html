Inputs


Text fields allow you to enter information and send it later, for example to the server. They are a very important element, so for us it is important that they are functional and at the same time nice. Our text fields work only after adding appropriate classes (without classes there are default). See examples below:

<!-- TEXT -->
<div class="field-text">
<input type="text" required/><label>Text</label>
</div>
<!-- /TEXT -->

<!-- PASSWORD -->
<div class="field-text">
<input type="password" required/><label>Password</label>
</div>
<!-- /PASSWORD -->

<!-- TELEPHONE -->
<div class="field-text">
<input type="tel" required/><label>Tel</label>
</div>
<!-- /TELEPHONE -->

<!-- TEXTAREA -->
<div class="field-text">
<textarea required rows="5"></textarea><label>Textarea</label>
<!-- /TEXTAREA -->

<!-- DISABLED -->
<div class="field-text">
<input type="text" disabled/><label>Disabled</label>
</div>
<!-- /DISABLED -->

<!-- FILE INPUT -->
<input type="file" name="file[]" id="file" class="inputfile inputfile" data-multiple-caption="{count} files selected"/>
<label for="file"> <span>Choose a file</span></label>
<!-- /FILE INPUT -->

<!-- DISABLED FILE INPUT -->
<input type="file" name="file[]" id="file1" class="inputfile inputfile" data-multiple-caption="{count} files selected" disabled/>
<label for="file1"> <span>Choose a file</span></label>
<!-- /DISABLED FILE INPUT -->

<!-- MULTIPLE FILE INPUT -->
<input type="file" name="file2[]" id="file2" class="inputfile inputfile" data-multiple-caption="{count} files selected"  multiple />
<label for="file2"> <span>Choose a files</span></label>
<!-- MULTIPLE FILE INPUT -->>
Note: Email field must have type="text"
